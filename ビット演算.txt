# bit演算
# bit集合

# ビットで集合Aを表す
A = {0,1,2,3,4}
A = 31（10進数） = 11111（2進数）

# ビットでAの部分集合Bを表す
B = {0, 1, 3}
B = 11（10進数） = 01011（2進数）

# 和集合はORで計算できる
A = {0,1}, B = {0, 3}
A+B = 0011 | 1001 = 1011 = {0,1,3}
3 | 9 = 11

# 積集合はANDで計算できる
A = {0,1}, B = {0, 3}
A*B = 0011 & 0101 = 0111 = {0}
3 & 9 = 1

# 補集合はすべてが1のビットとのXORで計算できる
U = {0,1,2,3}, A = {0,1}
U-A = 1111 ^ 0011 = 1100 = {2,3}
15 - 3 = 12
